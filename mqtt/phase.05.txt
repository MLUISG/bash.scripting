# node-red
# https://nodered.org/docs/getting-started/local

# remove any existing container
docker container run -d --name node_red -p 1880:1880 mosquitto sleep inf


# install node-red
docker container exec -i node_red /bin/bash  << 'eof'
  apt-get update &&
  apt-get install -y --no-install-recommends \
    npm \
    ;
  npm install -g --unsafe-perm node-red
eof

docker container commit mosquitto node_red

docker container run -d --name node_red -p 1880:1880 node_red sleep inf

docker container exec -it node_red /bin/bash

node-red

docker container run -d --name node_red -p 1880:1880 node_red node-red
